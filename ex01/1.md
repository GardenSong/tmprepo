```sh
node -v              # 현재 설치된 Node.js의 버전을 확인합니다.
cd ex01              # 'ex01' 폴더로 이동합니다.
node index.js        # 'index.js' 파일을 Node.js로 실행합니다.
node .               # 현재 폴더의 기본 실행 파일(index.js)을 Node.js로 실행합니다.
history              # 터미널에서 사용한 명령어의 기록을 출력합니다.

```
